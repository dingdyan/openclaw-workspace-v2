function Cycle(InitV, A){
  var P, Stack, result, U, V;
  result = NULL;
  Stack = [InitV, NULL];
  while (Stack != NULL) {
    V = Stack[0];
    P = & A[V];
    if (P != NULL) {
      U = P[0];
      reduced P = P[1];
      Stack = [U, Stack];
      P = & A[U];
      while (P != NULL) {
        if (P[0] == V) {
          reduced P = P[1];
          break;
        }
        P = & P[1];
      }
    }
    else {
      reduced Stack = Stack[1];
      result = [V, result];
    }
  }
  return result;
}
